# Web UI & SSE Delivery Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the server-rendered presentation surface and the live SSE stream for Breakbeat. Turn one frozen company anchor into a started Job (disambiguating at input), then read the Job's growing read models — Resolved Identity, Job-level Summary, scored Results, Exclusions — across four nunjucks pages. Page 1 of a running Job's Result list streams new and re-scored rows in over SSE, re-sorting by Match Score descending without violent reflow, announcing arrivals to screen-reader users, and freezing into a stable paginated snapshot at terminal. Every NULL reads honestly ("Unverified", "Unclassified"); nothing rides colour alone; WCAG 2.2 AA is the floor.

**Architecture:** Thin interface + presentation slice over Foundation's hexagonal layering (`interface → application → domain`, `infrastructure` implements `application`'s ports). Pure framework-free **view-models** (`src/presentation/`, the richest Vitest target); five **read-model ports** (`src/application/read/`) with Drizzle read adapters (`src/infrastructure/read/`); an ioredis **subscriber** + a worker-side **per-Result publish seam**; nunjucks **controllers/templates** + two **Lit islands** + an **SSE handler** (`src/interface/web/`). The UI imports no stage class, no `Stage` port, no write-side repository method — it reads its own purpose-built read models and consumes the SSE stream. It never computes or re-ranks Match Score.

**Tech Stack:** TypeScript, NestJS 11 (Express), nunjucks, htmx, Lit 3, Tailwind v4 (`@tailwindcss/cli`), ioredis, drizzle-orm/postgres-js, Zod, `@nestjs/config`. Vitest (unit, `*.test.ts`; single-adapter integration, `*.integration.test.ts` against compose Postgres/Redis); Playwright (`*.e2e.ts`, the two-process spine) with `@axe-core/playwright`. Biome + `tsc` + FTA gates.

```ts
export async function seedJob(sql: postgres.Sql, job: SeedJob): Promise<string> {
  const jobId = job.jobId ?? crypto.randomUUID();
  await sql`
    insert into jobs (id, anchor_kind, anchor_name, anchor_domain, anchor_provenance, state, created_at)
    values (${jobId}, ${job.anchorDomain ? "disambiguated" : "name_only"}, ${job.anchorName},

            ${job.state}, now())`;

    await sql`insert into resolved_identity (job_id, company_name, negative_boost, created_at)
              values (${jobId}, ${job.companyName}, '', now())`;
  }
  for (const r of job.results ?? []) {
    await sql`
      insert into results (id, job_id, url, normalized_url, title, snippet, match_score,
                           verification_status, content_type, sentiment, takeaway, status,
                           exclusion_code, exclusion_detail, published_date, source)

              ${r.matchScore}, ${r.verificationStatus ?? null}, ${r.contentType ?? null},
              ${r.sentiment ?? null}, ${r.takeaway ?? null}, ${r.status ?? "included"},
              ${r.exclusionCode ?? null}, ${r.exclusionDetail ?? null}, ${r.publishedDate ?? null}, 'tavily')`;
  }
  return jobId;
}
```

```ts
// src/presentation/null-readings.ts

/** The closed Verification status set written by the Verify stage; NULL = "hasn't been verified". */
export type VerificationStatus = "verified" | "uncertain";

/** The closed Content Type set written by the Classify stage; NULL = "hasn't been classified". */
export type ContentType =
  | "news_article"
  | "trade_publication"
  | "press_release"
  | "blog_post"
  | "newsletter"
  | "major_social_post"
  | "podcast"
  | "other";

const VERIFICATION_LABELS: Record<VerificationStatus, string> = {
  verified: "Verified",
  uncertain: "Uncertain match",
};

const CONTENT_TYPE_LABELS: Record<ContentType, string> = {
  news_article: "News article",
  trade_publication: "Trade publication",
  press_release: "Press release",
  blog_post: "Blog post",
  newsletter: "Newsletter",
  major_social_post: "Major social post",
  podcast: "Podcast",
  other: "Other",
};

/** Reading of `verification_status`: a real status → its label; NULL → "Unverified" (computed, never stored). */
export function readVerification(status: VerificationStatus | null): string {
  return status === null ? "Unverified" : VERIFICATION_LABELS[status];
}

/** Reading of `content_type`: a real type → its label; NULL → "Unclassified" ("other" is distinct, reads "Other"). */
export function readContentType(type: ContentType | string | null): string {
  if (type === null) return "Unclassified";
  return CONTENT_TYPE_LABELS[type as ContentType] ?? "Other";
}
```

```ts
// src/presentation/match-score.vm.ts

/**
 * The Match Score Indicator VM. `scored` carries the 0–100 numeric and an equal `widthPct`
 * (the bar fills Ink, structural — the template uses the Ink token, never a brand bright).
 * `unscored` is rendered as the "Unverified" reading by the row VM, INDEPENDENT of
 * verification_status (one NULL never implies the other — see match-score.vm.test.ts).
 */
export type ScoreVM = { kind: "scored"; numeric: number; widthPct: number } | { kind: "unscored" };

export function toScoreBar(matchScore: number | null): ScoreVM {
  if (matchScore === null) return { kind: "unscored" };
  const numeric = Math.min(100, Math.max(0, Math.round(matchScore)));
  return { kind: "scored", numeric, widthPct: numeric };
}
```

```ts
/** Mirrors Foundation's JobState union (confirm the exact export during execution). */
export type JobState = "pending" | "running" | "done" | "done_with_warnings" | "failed";

export type StatusTone = "neutral" | "info" | "success" | "warning" | "error";
export type StatusDotKind = "idle" | "pulse" | "solid";
export type StatusBadge = { label: string; tone: StatusTone; dotKind: StatusDotKind };

const BADGES: Record<JobState, StatusBadge> = {
  pending: { label: "Pending", tone: "neutral", dotKind: "idle" },
  running: { label: "Researching…", tone: "info", dotKind: "pulse" },
  done: { label: "Done", tone: "success", dotKind: "solid" },
  done_with_warnings: { label: "Done · warnings", tone: "warning", dotKind: "solid" },
  failed: { label: "Failed", tone: "error", dotKind: "solid" },
};

/**
 * JobState → badge. The label and dotKind carry the meaning for colour-blind users; `tone`
 * only selects the palette step (CSS stills the `pulse` under prefers-reduced-motion).
 */
export function toStatusBadge(state: JobState): StatusBadge {
  return BADGES[state];
}
```

```ts
export type ContentTypeGroup = "editorial" | "written" | "social" | "other";
export type ContentTypeChip = { group: ContentTypeGroup; iconKey: string; label: string };

const GROUP: Record<ContentType, ContentTypeGroup> = {
  news_article: "editorial",
  trade_publication: "editorial",
  press_release: "editorial",
  blog_post: "written",
  newsletter: "written",
  major_social_post: "social",
  podcast: "social",
  other: "other",

  news_article: "news",
  trade_publication: "trade",

  blog_post: "blog",
```

```ts
// src/presentation/format.vm.ts

const MONTHS = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];

const pad = (n: number) => String(n).padStart(2, "0");

/** Absolute day, "19 Aug 2024" (UTC). NULL → "—". */
export function formatDate(date: Date | null): string {
  if (date === null) return "—";
  return `${date.getUTCDate()} ${MONTHS[date.getUTCMonth()]} ${date.getUTCFullYear()}`;
}

/** Registrable host from a URL; strips scheme + leading www. Degrades to the raw string, never throws. */
export function formatSourceDomain(url: string): string {
  try {
    return new URL(url).host.replace(/^www\./, "");
  } catch {
    return url;
  }
}

export type RunTimeVM = { relative: string; title: string };

/** Relative "2h ago" with an absolute title "9 Jun 2026, 10:00". `now` injected for determinism. */
export function formatRunTime(when: Date, now: Date): RunTimeVM {
  const title = `${when.getUTCDate()} ${MONTHS[when.getUTCMonth()]} ${when.getUTCFullYear()}, ${pad(
    when.getUTCHours(),
  )}:${pad(when.getUTCMinutes())}`;
  const seconds = Math.floor((now.getTime() - when.getTime()) / 1000);
  let relative: string;
  if (seconds < 60) relative = "just now";
  else if (seconds < 3600) relative = `${Math.floor(seconds / 60)}m ago`;
  else if (seconds < 86_400) relative = `${Math.floor(seconds / 3600)}h ago`;
  else {
    const days = Math.floor(seconds / 86_400);
    relative = days === 1 ? "Yesterday" : `${days} days ago`;
  }
  return { relative, title };
}
```

```ts
import { type ContentType, readContentType } from "./null-readings";
import { toContentTypeChip, type ContentTypeGroup } from "./content-type.vm";

export type ContentTypeCount = { contentType: string | null; count: number };

export type FilterChip = {
  key: string;            // "all" | a ContentType | "unclassified"
  label: string;
  count: number;
  group: ContentTypeGroup | null; // null for the "All" chip
  selected: boolean;
  disabled: boolean;
};

/** Fixed canonical chip order (matches the read-model ordering + the mockup). */
const CANONICAL: ContentType[] = [
  "news_article",
  "trade_publication",
  "press_release",
  "blog_post",
  "newsletter",
  "major_social_post",
  "podcast",
];

/**
 * typeCounts (over ALL included rows) + the selected type → the ordered chip list. "All" first with
 * the total; one chip per canonical type (zero-count → `disabled`, never `selected`); an
 * "Unclassified" chip only when the NULL bucket count > 0.
 */
export function deriveChips(typeCounts: ContentTypeCount[], selectedType: string | null): FilterChip[] {
  const countOf = (key: string | null) => typeCounts.find((c) => c.contentType === key)?.count ?? 0;
  const total = typeCounts.reduce((sum, c) => sum + c.count, 0);

  const chips: FilterChip[] = [
    { key: "all", label: "All", count: total, group: null, selected: selectedType === null, disabled: false },
  ];

  for (const type of CANONICAL) {
    const count = countOf(type);
    chips.push({
      key: type,
      label: readContentType(type),
      count,
      group: toContentTypeChip(type).group,
      selected: selectedType === type,
      disabled: count === 0,
    });
  }

  const nullCount = countOf(null);
  if (nullCount > 0) {
    chips.push({
      key: "unclassified",
      label: "Unclassified",
      count: nullCount,
      group: "other",
      selected: selectedType === "unclassified",
      disabled: false,
    });
  }

  return chips;
}
```

```ts
// src/presentation/pagination.vm.ts

export type PageToken = number | "ellipsis";
export type PaginationVM = {
  page: number;            // clamped current page
  rangeLabel: string;      // "15–21 of 21" | "0 of 0"
  pages: PageToken[];      // windowed button list with ellipses
  prevDisabled: boolean;
  nextDisabled: boolean;
};

/**
 * Pure pagination maths. Clamps `page` into [1, totalPages]; "0 of 0" when empty; a windowed
 * page list (first 3 + ellipsis + last, or the full run when short). Both read-model lists
 * (Searches + Result rows) render through this single VM.
 */
export function paginate(total: number, requestedPage: number, pageSize: number): PaginationVM {
  const totalPages = Math.max(1, Math.ceil(total / pageSize));
  const page = Math.min(Math.max(1, requestedPage), totalPages);

  if (total === 0) {
    return { page: 1, rangeLabel: "0 of 0", pages: [1], prevDisabled: true, nextDisabled: true };
  }

  const from = (page - 1) * pageSize + 1;
  const to = Math.min(page * pageSize, total);

  return {
    page,
    rangeLabel: `${from}–${to} of ${total}`,
    pages: windowPages(page, totalPages),
    prevDisabled: page === 1,
    nextDisabled: page === totalPages,
  };
}

function windowPages(page: number, totalPages: number): PageToken[] {
  if (totalPages <= 4) return Array.from({ length: totalPages }, (_, i) => i + 1);
  const head = [1, 2, 3].filter((p) => p < totalPages);
  const around = [page - 1, page, page + 1].filter((p) => p > 3 && p < totalPages);
  const middle = [...new Set([...head, ...around])].sort((a, b) => a - b);
  const tokens: PageToken[] = [];
  let prev = 0;
  for (const p of middle) {

  tokens.push(totalPages);
```

```ts
  resultId: string;
  headline: string;

  sourceDomain: string;
  dateLabel: string;

  verification: string;            // the reading (independent of score)

/**

 * (one NULL never implies the other). One VM shape ⇒ an SSE-inserted row is identical to a
 * server-rendered row (story 24/25). The page link is always `/jobs/:jobId/page/:resultId`.
 */
export function toResultRowVM(row: ResultRowRead, opts: RowVMOptions): ResultRowVM {
  const excluded = row.exclusionCode != null;
  const vm: ResultRowVM = {
    resultId: row.resultId,
    headline: row.headline,
    pageHref: `/jobs/${row.jobId}/page/${row.resultId}`,
    sourceDomain: formatSourceDomain(row.url),
    dateLabel: formatDate(row.publishedDate),
    score: toScoreBar(row.matchScore),
    verification: readVerification(row.verificationStatus),
    contentType: toContentTypeChip(row.contentType),
    lowConfidence: row.verificationStatus === "uncertain",
    excluded,
  };

    vm.exclusionReason = toExclusionReason(row.exclusionCode, row.exclusionDetail ?? null, opts.resolveHeadline);
  }
  return vm;
}
```

```ts
// src/presentation/row-merge.ts
import type { ResultRowVM } from "./result-row.vm";

const scoreOf = (r: ResultRowVM): number => (r.score.kind === "scored" ? r.score.numeric : -1);

/**
 * Insert/update-by-id then re-sort by Match Score descending. The single invariant that keeps the
 * server-rendered page 1 and the live-mutated page 1 identical — it orders by the SAME key as the
 * `page === 1` read-model query: match_score DESC, NULLS LAST (unscored sink to -1), ties by
 * resultId. De-dupes by id (ADR 0006 row-level idempotency); pure (does not mutate `rows`).
 */
export function mergeRow(rows: readonly ResultRowVM[], incoming: ResultRowVM): ResultRowVM[] {
  const next = rows.filter((r) => r.resultId !== incoming.resultId);
  next.push(incoming);
  next.sort((a, b) => scoreOf(b) - scoreOf(a) || a.resultId.localeCompare(b.resultId));
  return next;
}
```

```ts
// src/infrastructure/read/results.read.ts
import { sql } from "drizzle-orm";
import type { PostgresJsDatabase } from "drizzle-orm/postgres-js";
import type {
  ResultsReadPort,
  ResultsView,
  ResultsReadOptions,
  ResultRowRead,
  ExcludedRowRead,
  ContentTypeCount,
} from "../../application/read/ports/results.read-port";

type Db = PostgresJsDatabase<Record<string, never>>;
type RawRow = {
  id: string;
  job_id: string;
  title: string;
  url: string;
  published_date: Date | null;
  match_score: number | null;
  verification_status: ResultRowRead["verificationStatus"];
  content_type: string | null;
  exclusion_code: string | null;
  exclusion_detail: string | null;
};

const toRow = (r: RawRow): ResultRowRead => ({
  resultId: r.id,
  jobId: r.job_id,
  headline: r.title,
  url: r.url,
  publishedDate: r.published_date ? new Date(r.published_date) : null,
  matchScore: r.match_score === null ? null : Number(r.match_score),
  verificationStatus: r.verification_status,
  contentType: r.content_type,
});

/**
 * The load-bearing read. Partitions included|excluded; the included slice is filtered, ordered
 * `match_score DESC NULLS LAST, id` (the page-1 invariant the Lit re-sort matches — ADR 0007),
 * and paginated; typeCounts is the over-all-included aggregate (incl. the NULL bucket). Read-only.
 */
export class ResultsDrizzleRead implements ResultsReadPort {
  constructor(private readonly db: Db) {}

  async read(jobId: string, opts: ResultsReadOptions): Promise<ResultsView> {
    const offset = (opts.page - 1) * opts.pageSize;
    const typeFilter = opts.contentType == null ? sql`` : sql`and content_type is not distinct from ${opts.contentType}`;

    const included = await this.db.execute<RawRow>(sql`
      select id, job_id, title, url, published_date, match_score, verification_status, content_type,
             exclusion_code, exclusion_detail
      from results
      where job_id = ${jobId} and status = 'included' ${typeFilter}
      order by match_score desc nulls last, id
      limit ${opts.pageSize} offset ${offset}`);

    const includedTotalRows = await this.db.execute<{ total: number }>(sql`
      select count(*)::int as total from results
      where job_id = ${jobId} and status = 'included' ${typeFilter}`);

    const counts = await this.db.execute<{ content_type: string | null; count: number }>(sql`
      select content_type, count(*)::int as count from results
      where job_id = ${jobId} and status = 'included'
      group by content_type`);

    const excluded = await this.db.execute<RawRow>(sql`
      select id, job_id, title, url, published_date, match_score, verification_status, content_type,
             exclusion_code, exclusion_detail
      from results
      where job_id = ${jobId} and status = 'excluded'
      order by match_score desc nulls last, id`);

    return {
      included: [...included].map(toRow),
      includedTotal: [...includedTotalRows][0]?.total ?? 0,
      typeCounts: [...counts].map((c): ContentTypeCount => ({ contentType: c.content_type, count: c.count })),
      excluded: [...excluded].map((r): ExcludedRowRead => ({
        ...toRow(r),
        exclusionCode: r.exclusion_code as ExcludedRowRead["exclusionCode"],
        exclusionDetail: r.exclusion_detail,
      })),
      excludedTotal: [...excluded].length,
    };
  }

  async readOne(jobId: string, resultId: string): Promise<ResultRowRead | ExcludedRowRead | null> {
    const rows = await this.db.execute<RawRow>(sql`
      select id, job_id, title, url, published_date, match_score, verification_status, content_type,
             exclusion_code, exclusion_detail
      from results where job_id = ${jobId} and id = ${resultId} limit 1`);
    const r = [...rows][0];
    if (!r) return null;
    if (r.exclusion_code != null) {
      return { ...toRow(r), exclusionCode: r.exclusion_code as ExcludedRowRead["exclusionCode"], exclusionDetail: r.exclusion_detail };
    }
    return toRow(r);
  }
}
```

```ts
// src/infrastructure/events/redis-event.subscriber.ts
import Redis from "ioredis";
import { z } from "zod";
import type { JobEventSubscriberPort, JobNudge } from "../../application/read/ports/job-event-subscriber.port";
import { jobChannel } from "./redis-event.publisher";

const nudgeSchema = z.object({
  jobId: z.string(),
  kind: z.enum(["result", "status"]),
  id: z.string().optional(),
});

/**
 * ADR 0006 subscriber. Each subscribe() opens a DEDICATED ioredis connection in subscriber mode
 * (a subscribed connection can't run other commands — kept separate from the publisher/BullMQ
 * connection). Tolerant parse: a malformed/duplicate message is dropped, never thrown. The
 * returned thunk unsubscribes and releases the connection on stream close — no leaked subscriptions.
 */
export class RedisEventSubscriber implements JobEventSubscriberPort {
  constructor(private readonly redisUrl: string) {}

  async subscribe(jobId: string, onNudge: (n: JobNudge) => void): Promise<() => Promise<void>> {
    const channel = jobChannel(jobId);
    const connection = new Redis(this.redisUrl, { lazyConnect: false });

    connection.on("message", (incomingChannel, payload) => {
      if (incomingChannel !== channel) return;
      try {
        const parsed = nudgeSchema.safeParse(JSON.parse(payload));
        if (parsed.success) onNudge(parsed.data);
      } catch {
        // malformed JSON — drop it (harmless by design; the next nudge or a reconnect re-reads)
      }
    });

    await connection.subscribe(channel);

    return async () => {
      try {
        await connection.unsubscribe(channel);
      } finally {
        connection.disconnect();
      }
    };
  }
}
```

> Inject `REDIS_URL` (or the shared connection config) via DI in Task 26. The DI provider builds one subscriber instance; each `subscribe()` opens its own short-lived connection scoped to one SSE stream.

```ts
// src/interface/web/brand-search.controller.ts
import { Controller, Get, Inject, Query } from "@nestjs/common";
import { BRAND_SEARCH_PORT, type BrandSearchPort } from "../../application/resolve/ports/brand-search.port";
import { READ_CONFIG, type ReadConfig } from "../../application/read/read-config";

@Controller("brand-search")
export class BrandSearchController {
  constructor(
    @Inject(BRAND_SEARCH_PORT) private readonly brandSearch: BrandSearchPort,
    @Inject(READ_CONFIG) private readonly config: ReadConfig,
  ) {}

  /** Autocomplete + options-list backing. Reuses Resolve's BrandSearchPort — no second client. */
  @Get()
  async search(@Query("q") q = ""): Promise<{ brandId: string | null; name: string; domain: string | null }[]> {
    const query = q.trim();
    if (query.length < this.config.autocompleteMinChars) return [];
    const hits = await this.brandSearch.search(query); // [] on empty/failure (never throws)
    return hits.map((h) => ({ brandId: h.brandId, name: h.name, domain: h.domain }));
  }
}
```

```ts
import { Get, Param, Query, Render, Req } from "@nestjs/common";
import type { Request } from "express";
import { RESOLVED_IDENTITY_READ_PORT, type ResolvedIdentityReadPort } from "../../application/read/ports/resolved-identity.read-port";
import { SUMMARY_READ_PORT, type SummaryReadPort } from "../../application/read/ports/summary.read-port";
import { RESULTS_READ_PORT, type ResultsReadPort } from "../../application/read/ports/results.read-port";
import { toResultRowVM } from "../../presentation/result-row.vm";
import { deriveChips } from "../../presentation/chips.vm";
import { paginate } from "../../presentation/pagination.vm";
import { toStatusBadge } from "../../presentation/job-status.vm";

// constructor injects the three read ports + READ_CONFIG + Foundation's JobRepository (for state/anchor).

@Get("jobs/:id")
async resultPage(@Param("id") jobId: string, @Query("type") type: string | undefined, @Req() req: Request) {
  const job = await this.jobRepo.findById(jobId);     // Foundation's read for state + anchor
  if (!job) { /* render 404 */ }
  const now = new Date();
  const size = this.cfg.resultsPageSize;
  const contentType = type ?? null;

  const [profile, summary, results] = await Promise.all([
    this.identity.findByJobId(jobId),
    this.summary.findByJobId(jobId),
    this.results.read(jobId, { page: 1, pageSize: size, contentType }),
  ]);

  const winnerHeadline = (rid: string) =>
    results.included.find((r) => r.resultId === rid)?.headline ?? null;
  const rowVMs = results.included.map((r) => toResultRowVM(r, { now, resolveHeadline: winnerHeadline }));
  const excludedVMs = results.excluded.map((r) => toResultRowVM(r, { now, resolveHeadline: winnerHeadline }));

  const ctx = {
    title: profile?.companyName ?? job.anchor.name ?? "Result",
    nav: "searches",
    jobId,
    state: job.state,
    badge: toStatusBadge(job.state),
    profile: profile ?? { companyName: job.anchor.name ?? jobId, hasIdentity: false, ownDomains: [], handles: [], tags: [], alsoKnownAs: [], collisions: [], description: null },
    summary: summary.text,
    chips: deriveChips(results.typeCounts, contentType),
    rows: rowVMs,
    excluded: excludedVMs,
    excludedTotal: results.excludedTotal,
    pagination: paginate(results.includedTotal, 1, size),
    isTerminal: ["done", "done_with_warnings", "failed"].includes(job.state),
    isFailed: job.state === "failed",
    isEmpty: results.includedTotal === 0,
    selectedType: contentType,
  };

  // htmx filter/pagination round-trips swap only the results-list partial:
  if (req.header("HX-Request")) return this.view.render("partials/results-list.njk", ctx);
  return this.view.render("result.njk", ctx);
}
```

```ts
// src/interface/web/assets/bb-autocomplete.ts
import { LitElement, html, nothing } from "lit";
import { customElement, state } from "lit/decorators.js";

type Suggestion = { brandId: string | null; name: string; domain: string | null };

/**
 * Light-DOM Lit island wrapping the homepage form's input. Debounced GET /brand-search; renders a
 * keyboard-navigable role="listbox" (Float shadow); picking fills the hidden disambiguated fields
 * (provenance="picked") and submits the form. Renders into light DOM so the surrounding nunjucks
 * form + htmx still own the markup; the element only manages the suggestion popover + keyboard state.
 */
@customElement("bb-autocomplete")
export class BbAutocomplete extends LitElement {
  @state() private suggestions: Suggestion[] = [];
  @state() private open = false;
  @state() private activeIndex = -1;

  private input!: HTMLInputElement;
  private debounceMs = 180;
  private minChars = 2;
  private timer: number | undefined;

  // Render into light DOM so /public/tailwind.css applies and the form is the real submit target.
  protected createRenderRoot(): HTMLElement {
    return this;
  }

  connectedCallback(): void {
    super.connectedCallback();
    this.minChars = Number(this.dataset.minChars ?? 2);
    this.debounceMs = Number(this.dataset.debounceMs ?? 180);
    this.input = this.querySelector("#q") as HTMLInputElement;
    this.input.setAttribute("role", "combobox");
    this.input.setAttribute("aria-autocomplete", "list");
    this.input.setAttribute("aria-expanded", "false");
    this.input.setAttribute("aria-controls", "bb-ac-listbox");
    this.input.addEventListener("input", this.onInput);
    this.input.addEventListener("keydown", this.onKeydown);
  }

  disconnectedCallback(): void {
    this.input?.removeEventListener("input", this.onInput);
    this.input?.removeEventListener("keydown", this.onKeydown);
    super.disconnectedCallback();
  }

  private onInput = () => {
    window.clearTimeout(this.timer);
    const q = this.input.value.trim();
    if (q.length < this.minChars) {
      this.close();
      return;
    }
    this.timer = window.setTimeout(() => void this.fetchSuggestions(q), this.debounceMs);
  };

  private async fetchSuggestions(q: string): Promise<void> {
    try {
      const res = await fetch(`/brand-search?q=${encodeURIComponent(q)}`);
      this.suggestions = res.ok ? ((await res.json()) as Suggestion[]) : [];
    } catch {
      this.suggestions = [];
    }
    this.open = this.suggestions.length > 0;
    this.activeIndex = -1;
    this.input.setAttribute("aria-expanded", String(this.open));
  }

  private onKeydown = (e: KeyboardEvent) => {
    if (!this.open) return;
    if (e.key === "ArrowDown") { e.preventDefault(); this.move(1); }
    else if (e.key === "ArrowUp") { e.preventDefault(); this.move(-1); }
    else if (e.key === "Enter" && this.activeIndex >= 0) { e.preventDefault(); this.pick(this.suggestions[this.activeIndex]); }
    else if (e.key === "Escape") { e.preventDefault(); this.close(); }
  };

  private move(delta: number): void {
    const n = this.suggestions.length;
    this.activeIndex = (this.activeIndex + delta + n) % n;
    this.input.setAttribute("aria-activedescendant", `bb-ac-opt-${this.activeIndex}`);
  }

  private close(): void {
    this.open = false;
    this.activeIndex = -1;
    this.input.setAttribute("aria-expanded", "false");
    this.input.removeAttribute("aria-activedescendant");
  }

  private pick(s: Suggestion): void {
    const form = this.querySelector("form") as HTMLFormElement;
    (form.querySelector('[name="brandId"]') as HTMLInputElement).value = s.brandId ?? "";
    (form.querySelector('[name="domain"]') as HTMLInputElement).value = s.domain ?? "";
    (form.querySelector('[name="provenance"]') as HTMLInputElement).value = "picked";
    this.input.value = s.name;
    this.close();
    form.requestSubmit(); // native POST /jobs → redirect to the Result page
  }

  protected render() {
    if (!this.open) return nothing;
    return html`
      <ul id="bb-ac-listbox" role="listbox" aria-label="Company suggestions"
          class="mt-1 list-none rounded-md border border-border bg-surface p-1 shadow-float">
        ${this.suggestions.map(
          (s, i) => html`
            <li id="bb-ac-opt-${i}" role="option" aria-selected=${i === this.activeIndex}
                class="flex cursor-pointer items-center gap-3 rounded-md px-3 py-2 ${i === this.activeIndex ? "bg-raised" : ""}"
                @mousedown=${(e: Event) => { e.preventDefault(); this.pick(s); }}>
              <span class="font-medium">${s.name}</span>
              ${s.domain ? html`<span class="text-sm text-ink-toned">· ${s.domain}</span>` : nothing}
            </li>`,
        )}
      </ul>`;
  }
}
```

> `requestSubmit()` runs native validation + the form's normal POST (which 302s to the Result page). The listbox renders in light DOM so `/public/tailwind.css` styles it. The `@playwright/test` `page.route` stub makes the suggestion set deterministic.

```ts
// src/interface/web/assets/row-render.ts
import type { ResultRowVM } from "../../../presentation/result-row.vm";

const escapeHtml = (s: string): string =>
  s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;");

/**
 * Pure ResultRowVM → row <li> HTML, MIRRORING partials/result-row.njk so a streamed row is
 * byte-identical to a server-rendered row (story 24/25). All interpolated values are escaped
 * (anti-echo — no raw model/page text is ever injected as HTML).
 */
export function renderRow(row: ResultRowVM): string {
  const score = row.score.kind === "scored" ? row.score.numeric : -1;
  const meta =
    row.excluded && row.exclusionReason
      ? `<span class="text-ink-dimmed">·</span><span class="exreason inline-flex items-center gap-1 rounded-pill border border-border bg-sunken px-2 py-0.5 text-xs font-medium">${escapeHtml(row.exclusionReason)}</span>`
      : `<span class="text-ink-dimmed">·</span><span>${escapeHtml(row.contentType.label)}</span>${
          row.verification === "Verified" ? `<span class="text-ink-dimmed">·</span><span class="text-green-text">✓ Verified</span>` : ""
        }`;
  const scoreCol =
    row.score.kind === "scored"
      ? `<span class="score flex w-[88px] flex-col items-end gap-1.5"><span class="score__num text-sm font-semibold tabular-nums">${row.score.numeric}<span class="ml-0.5 text-[0.6875rem] font-medium text-ink-dimmed">/100</span></span><span class="score__bar h-[3px] w-full overflow-hidden rounded-sm bg-border"><span class="block h-full rounded-sm bg-ink" style="width:${row.score.widthPct}%"></span></span></span>`
      : `<span class="score text-[0.8125rem] text-ink-dimmed">${escapeHtml(row.verification)}</span>`;

  return `<li data-result-id="${escapeHtml(row.resultId)}" data-score="${score}">
    <a class="row grid grid-cols-[36px_1fr_auto] items-center gap-4 rounded-md border-t border-border px-3 py-[15px] no-underline text-ink hover:bg-raised" href="${escapeHtml(row.pageHref)}">
      <span class="tile grid h-9 w-9 flex-none place-items-center rounded-[9px] tint-${row.contentType.group}" title="${escapeHtml(row.contentType.label)}" aria-label="${escapeHtml(row.contentType.label)}"><span aria-hidden="true">•</span></span>
      <span class="row__main min-w-0">
        <p class="row__title m-0 mb-1 text-[1.0625rem] font-semibold leading-[1.3]${row.lowConfidence || row.excluded ? " text-ink-toned" : ""}">${escapeHtml(row.headline)}</p>

      </span>
      ${scoreCol}
    </a>
  </li>`;
}
```

```ts
// src/interface/web/assets/bb-result-stream.ts
import { LitElement } from "lit";
import { customElement } from "lit/decorators.js";
import { mergeRow } from "../../../presentation/row-merge";
import { renderRow } from "./row-render";
import type { ResultRowVM } from "../../../presentation/result-row.vm";

type ResultFrame = { type: "result"; row: ResultRowVM };
type StatusFrame = { type: "status"; state: string; badgeHtml: string };

const TERMINAL = new Set(["done", "done_with_warnings", "failed"]);

/**
 * Light-DOM island over the page-1 results list. Owns the EventSource; mutates ONLY the page-1 DOM
 * (ADR 0007). result → mergeRow re-sort + reconcile (≤200ms entrance / drop below the pageSize
 * cutoff); status → badge update + terminal close. Coalesced ARIA live announcements;
 * prefers-reduced-motion → instant. No SSE replay (page 1 was server-rendered).
 */
@customElement("bb-result-stream")
export class BbResultStream extends LitElement {
  private source?: EventSource;
  private rows: ResultRowVM[] = [];
  private pageSize = 25;
  private list!: HTMLUListElement;
  private live!: HTMLElement;
  private announceTimer?: number;
  private pendingArrivals = 0;
  private reduced = false;

  protected createRenderRoot(): HTMLElement {
    return this;
  }

  connectedCallback(): void {
    super.connectedCallback();
    const jobId = this.dataset.jobId as string;
    this.pageSize = Number(this.dataset.pageSize ?? 25);
    this.reduced = window.matchMedia("(prefers-reduced-motion: reduce)").matches;
    this.list = this.querySelector("ul.results") as HTMLUListElement;
    this.live = this.ensureLiveRegion();
    if (!this.list) return; // empty/failed/page-2+ render has no live list — nothing to stream

    // Seed `rows` from the server-rendered page-1 DOM so mergeRow keeps order without a replay.
    this.rows = [...this.list.querySelectorAll<HTMLLIElement>("li[data-result-id]")].map((li) => this.readRowFromDom(li));

    // Only page 1 streams. If a `page` query param > 1 is present, do not open the EventSource.
    if (new URLSearchParams(location.search).get("page") not_in [null, "1"]) return;
    this.source = new EventSource(`/jobs/${jobId}/stream`);
    this.source.addEventListener("result", (e) => this.onResult(JSON.parse((e as MessageEvent).data) as ResultRowVM));
    this.source.addEventListener("status", (e) => this.onStatus(JSON.parse((e as MessageEvent).data) as StatusFrame));
  }

  disconnectedCallback(): void {
    this.source?.close();
    super.disconnectedCallback();
  }

  private onResult(vm: ResultRowVM): void {
    const existing = this.rows.some((r) => r.resultId === vm.resultId);
    this.rows = mergeRow(this.rows, vm).slice(0, this.pageSize); // page-1 cutoff: DOM stays bounded
    this.reconcile(vm.resultId, !existing);
    if (!existing) this.queueArrivalAnnouncement();
  }

  private onStatus(frame: StatusFrame): void {
    const badge = document.querySelector(".status");
    if (badge && frame.badgeHtml) badge.outerHTML = frame.badgeHtml;
    if (TERMINAL.has(frame.state)) {
      this.source?.close();
      this.announce(`Job ${frame.state.replace(/_/g, " ")}.`);
    }
  }

  /** Rebuild the list DOM from the (already-sorted, capped) rows; flag the changed row's entrance. */
  private reconcile(changedId: string, isNew: boolean): void {
    this.list.innerHTML = this.rows.map((r) => renderRow(r)).join("");
    if (isNew && !this.reduced) {
      this.list.querySelector(`li[data-result-id="${changedId}"]`)?.classList.add("row--new");
    }
  }

  private queueArrivalAnnouncement(): void {
    this.pendingArrivals += 1;
    window.clearTimeout(this.announceTimer);
    this.announceTimer = window.setTimeout(() => {
      this.announce(`${this.pendingArrivals} new result${this.pendingArrivals === 1 ? "" : "s"} arrived.`);
      this.pendingArrivals = 0;
    }, 600); // coalesce — not one announcement per row
  }

  private announce(message: string): void {
    this.live.textContent = message;
  }

  private ensureLiveRegion(): HTMLElement {
    let region = this.querySelector<HTMLElement>("[data-bb-live]");
    if (!region) {
      region = document.createElement("div");
      region.setAttribute("data-bb-live", "");
      region.setAttribute("aria-live", "polite");
      region.className = "sr-only";
      this.prepend(region);
    }
    return region;
  }

  private readRowFromDom(li: HTMLLIElement): ResultRowVM {
    // Minimal reconstruction for ordering; the server-rendered DOM is authoritative for display
    // until a nudge replaces a row. score is read from data-score for the merge key.
    const score = Number(li.dataset.score ?? -1);
    return {
      resultId: li.dataset.resultId as string,
      headline: li.querySelector(".row__title")?.textContent?.trim() ?? "",
      pageHref: (li.querySelector("a.row") as HTMLAnchorElement)?.getAttribute("href") ?? "#",
      sourceDomain: li.querySelector(".src")?.textContent?.trim() ?? "",
      dateLabel: "",
      score: score >= 0 ? { kind: "scored", numeric: score, widthPct: score } : { kind: "unscored" },
      verification: "",
      contentType: { group: "other", iconKey: "other", label: "" },
      lowConfidence: li.querySelector(".row__title")?.classList.contains("text-ink-toned") ?? false,
      excluded: false,
    };
  }
}
```

```ts
  return page.evaluate(
    async ({ jobId, durationMs }) => {
      const res = await fetch(`/jobs/${jobId}/stream`, { headers: { Accept: "text/event-stream" } });
      const reader = res.body!.getReader();
      const decoder = new TextDecoder();
      let buf = "";

        if (done) break;
        buf += decoder.decode(value, { stream: true });

      reader.cancel();
      return buf;

  );
```

```ts
// src/interface/web/sse.controller.ts
import { Controller, Get, Inject, Param, Req, Res } from "@nestjs/common";
import type { Request, Response } from "express";
import { JOB_EVENT_SUBSCRIBER_PORT, type JobEventSubscriberPort, type JobNudge } from "../../application/read/ports/job-event-subscriber.port";
import { RESULTS_READ_PORT, type ResultsReadPort } from "../../application/read/ports/results.read-port";
import { toResultRowVM } from "../../presentation/result-row.vm";
import { toStatusBadge } from "../../presentation/job-status.vm";
import { renderStatusBadge, createViewEnv } from "./view-render";
// import { JobRepository } from <Foundation> — to read current Job state for status nudges

const TERMINAL = new Set(["done", "done_with_warnings", "failed"]);

@Controller()
export class SseController {
  private readonly view = createViewEnv();

  constructor(
    @Inject(JOB_EVENT_SUBSCRIBER_PORT) private readonly subscriber: JobEventSubscriberPort,
    @Inject(RESULTS_READ_PORT) private readonly results: ResultsReadPort,
    // private readonly jobRepo: JobRepository,  // Foundation's read for current state
  ) {}

  /** The only live surface. Subscribe → on each id-only nudge, re-read Postgres → emit a frame. */
  @Get("jobs/:id/stream")
  async stream(@Param("id") jobId: string, @Req() req: Request, @Res() res: Response): Promise<void> {
    res.writeHead(200, {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache, no-transform",
      Connection: "keep-alive",
      "X-Accel-Buffering": "no", // defeat proxy buffering
    });
    res.write(": connected\n\n"); // initial comment; metrics hook: activeConnections++ (PRD 8)

    const now = new Date();
    const heartbeat = setInterval(() => res.write(": keep-alive\n\n"), 25_000); // defeat idle proxies

    const onNudge = async (nudge: JobNudge): Promise<void> => {
      try {
        if (nudge.kind === "result" && nudge.id) {
          const row = await this.results.readOne(jobId, nudge.id); // content from Postgres at emit time
          if (!row) return;
          const vm = toResultRowVM(row, { now: new Date() });
          this.send(res, "result", vm); // metrics hook: messagesSent++ (PRD 8)
        } else if (nudge.kind === "status") {
          const job = await this.jobRepo.findById(jobId);
          if (!job) return;
          const badge = toStatusBadge(job.state);
          this.send(res, "status", { state: job.state, badgeHtml: renderStatusBadge(this.view, badge) });
          if (TERMINAL.has(job.state)) await close(); // terminal → end the stream
        }
      } catch {
        // a malformed/duplicate nudge is harmless — the next nudge or a reconnect re-reads
      }
    };

    const unsubscribe = await this.subscriber.subscribe(jobId, (n) => void onNudge(n));

    const close = async (): Promise<void> => {
      clearInterval(heartbeat);
      await unsubscribe();        // no leaked Redis subscription
      res.end();                  // metrics hook: activeConnections-- (PRD 8)
    };

    req.on("close", () => void close()); // client disconnect
  }

  private send(res: Response, event: string, data: unknown): void {
    res.write(`event: ${event}\ndata: ${JSON.stringify(data)}\n\n`);
  }
}
```
